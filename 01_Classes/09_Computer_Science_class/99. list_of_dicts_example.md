```python
  
def main():

    pets = [
    {"name": "Biscuit", "type": "dog"},
    {"name": "Miso", "type": "cat"},
    {"name": "Rex", "type": "dog"},
    {"name": "Nova", "type": "cat"},
    {"name": "Tweety", "type": "bird"}
]
    pets_by_type = {}

    for pet in pets: #loop through pets and read each pet
        type_of_pet = pet['type'] #find type of pet
        current_list = pets_by_type.get(type_of_pet, []) #add pet_type to list
        current_list.append(pet['name']) #add pet_name to list
        pets_by_type[type_of_pet] = current_list #put list together with pet_type and pet_name
    print(pets_by_type) #print the list of dicts

if __name__ == "__main__":
    main() #ensure main() only runs when directly called
```
