def filter_big(orders):
    '''returns a list of big orders with total'''
    return [order for order in orders if order['total'] >= 100]

def sum_totals(orders):
    '''returns sum total of orders'''
    return sum(order['total'] for order in orders)
  
def main():

    #sample list
    sample_orders = [
        {'total': 50, 'id': 1},
        {'total': 150, 'id': 2},
        {'total': 200, 'id': 3},
        {'total': 10, 'id': 4}
    ]

    #process data
    big_orders = filter_big(sample_orders)
    total_sum = sum_totals(big_orders)

    #print the result
    print(f'The sum of big orders >= 100 is: {total_sum}')

if __name__ == '__main__':

    main()
